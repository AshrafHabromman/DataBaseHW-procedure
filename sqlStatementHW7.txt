guest_ssn number(9) not null primary key ,

have_Covid19 varchar(1), 
age int not null ,
guest_name varchar(20) not null );





create or replace PROCEDURE GET_DATA_BY_SSN 
(
  GUEST_SSN_IN IN   HOTEL_GUEST.GUEST_SSN%TYPE ,
  GUEST_NAME_OUT OUT HOTEL_GUEST.GUEST_NAME%TYPE,
  AGE_OUT OUT HOTEL_GUEST.AGE%TYPE,
  HAVE_COVID19_OUT OUT HOTEL_GUEST.HAVE_COVID19%TYPE
) AS 
BEGIN
SELECT GUEST_NAME, AGE,HAVE_COVID19 INTO GUEST_NAME_OUT ,AGE_OUT,HAVE_COVID19_OUT
FROM HOTEL_GUEST
WHERE HOTEL_GUEST.GUEST_SSN = GUEST_SSN_IN ;
END GET_DATA_BY_SSN;
